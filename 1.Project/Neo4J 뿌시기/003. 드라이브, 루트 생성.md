@Validated  
@RestController  
@RequestMapping("/api/v1")  
class GraphfsApi(  
    private val foldersDataAccess: FoldersDataAccess,  
    private val drivesDataAccess: DrivesDataAccess,  
) : Controller {  
    @PostMapping(  
	    value = ["/drives"],  
	    consumes = [MediaType.APPLICATION_JSON_VALUE],  
	    produces = [MediaType.APPLICATION_JSON_VALUE],  
	)  
	suspend fun createDrive(): ResponseEntity<Unit> {  
	    val root = Folder(  
	        id = UUID.randomUUID().toString(),  
	        name = "child-1",  
	        depth = 1,  
	        creator = "tester",  
	        createdAt = OffsetDateTime.now(),  
	        children = emptyList(),  
	        resources = emptyList(),  
	    )  
	  
	    val drive = Drive(  
	        id = UUID.randomUUID().toString(),  
	        name = "DRIVE",  
	        projectId = "project",  
	        domain = "VIDEO",  
	        children = listOf(root),  
	        creator = "System",  
	        createdAt = OffsetDateTime.now(),  
	    )  
	  
	    foldersDataAccess.save(root).awaitSingle()  
	    drivesDataAccess.save(drive).awaitSingle()  
	  
	    return ResponseEntity.created(  
	        URI.create("$LOCATION_PREFIX/${drive.id}")  
	    )  
	        .build()  
	}
}
