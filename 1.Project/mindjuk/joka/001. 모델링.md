### 용어사전
- Media: 사진, 영상 등을 묶는 논리적 리소스
### 모델링
```typescript
interface Media {
	id: string;
	description: string; // 이름
	state: State;
	content?: Content;
	isFavorite: boolean;
	created: Actioned;
}

type State = "DRAFT" | "COMPLETE";

interface Actioned {
	at: string; // 2026-01-01 00:00:00
	by: By;
}

interface By {
	id: string;
	name: string;
	email: string;
}

interface Content {
	url: string;
	size: number;
	eTag: string;
	mimeType: string; // image/png, video/mp4, etc...
	thumbnail?: Thumbnail;
}

interface Thumbnail {
	url: string;
	size: number;
	eTag: string;
	mimeType: string; // image/jpeg, image/gif
	blurhash: string; // 일단은 blurhash에 의존해도 되는걸로 합의함!
}
```
