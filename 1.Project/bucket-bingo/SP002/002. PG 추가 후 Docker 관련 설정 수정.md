```yaml
spring:  
  profiles:  
    active: ${ACTIVE_PROFILE:local}  
  data:  
    mongodb:  
      username: root  
      password: bucket-bingo-api  
      host: host.docker.internal  
      port: 27017  
      database: local  
      authentication-database: admin  
  datasource:  
    driver-class-name: org.postgresql.Driver  
    url: jdbc:postgresql://host.docker.internal:5432/postgres  
    username: postgres  
    password: my-first-nest-pw  
  jpa:  
    hibernate:  
      ddl-auto: create #create update none  
      naming:  
        physical-strategy: org.hibernate.boot.model.naming.PhysicalNamingStrategyStandardImpl  
    show-sql: true  
  
server:  
  port: 33728
```

```yaml
version: '3.9'  
services:  
  bucket-bingo:  
    image: "bucket-bingo-api:0.0.1"  
    build:  
      context: .  
    depends_on:  
      mongodb:  
        condition: service_healthy  
      postgresql:  
        condition: service_healthy  
    container_name: bucket-bingo-api  
    networks:  
      - ineg  
    ports:  
      - '33728:33728'  
    restart: always  
  
  mongodb:  
    image: mongo:4.2  
    restart: always  
    container_name: bucket-bingo-mongo  
    networks:  
      - ineg  
    ports:  
      - '27017:27017'  
    expose:  
      - 27017  
    volumes:  
      - /tmp/mongodb3:/data/db  
    environment:  
      - MONGO_INITDB_ROOT_PASSWORD=bucket-bingo-api  
      - MONGO_INITDB_ROOT_USERNAME=root  
      - MONGO_INITDB_DATABASE=local  
    healthcheck:  
      test: echo 'db.runCommand("ping").ok'  
      interval: 3s  
      timeout: 3s  
      retries: 3  
      start_period: 5s  
  
  postgresql:  
    image: postgres:14.6  
    container_name: bucket-bingo-pg  
    networks:  
      - ineg  
    expose:  
      - 5432  
    ports:  
      - '5432:5432'  
    volumes:  
      - /tmp/pgdata:/var/lib/postgresql/data  
    environment:  
      - POSTGRES_PASSWORD=my-first-nest-pw  
      - TZ=Asia/Seoul  
    healthcheck:  
      test: [ "CMD-SHELL", "pg_isready", "-d", "db_prod" ]  
      interval: 3s  
      timeout: 3s  
      retries: 3  
      start_period: 5s  
  
networks:  
  ineg:
```
